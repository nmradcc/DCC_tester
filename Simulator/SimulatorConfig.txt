# DCC Command Station RPC Simulator Configuration
# ================================================

# Serial port to listen on (requires virtual COM port pair)
# On Windows: Use com0com to create a virtual port pair (e.g., COM9 <-> COM10)
# Scripts connect to COM9, simulator listens on COM10
serial_port=COM10

# Serial communication settings
baudrate=115200
timeout=2

# Response mode:
#   "default" - Use built-in response library
#   "replay" - Learn from verbose log file
#   "scenario" - Load specific test scenario
response_mode=default

# Path to verbose log file (for replay mode)
# Set to None or leave empty for default mode
log_file=None

# Path to scenario JSON file (for scenario mode)
scenario_file=None

# Enable logging of all RPC traffic
enable_logging=true

# Directory for simulator logs
log_directory=simulator_logs

# Simulate realistic timing delays
# Set to false for instant responses (useful for debugging)
simulate_timing=true

# Verbose console output
verbose=true
